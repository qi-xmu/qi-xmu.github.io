    # Python语言中，会明确地把代表自身实例的self作为第一个参数传入
    # init()方法接收参数(一定要有init()函数)

# 利用父类构造函数
class parent_class:
    def __init__(self, param1):
        pass
    def par_fun():
        pass
class son_class(parent_class):
    def __init__(self, param2):
        pass
    def son_fun():
        pass
    pass
# 调用
son_cla1 = son_class(param2)
son_cla1.son_fun()		# 调用子类的方法
son_cla1.par_fun()		# 调用父类的方法
